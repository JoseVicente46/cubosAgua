 19    def genStrCubos(self):
 20        """
 21        Esta funcion crea la estructura de los cubos, dependiendo de si estan llenos o vacios los pone de una forma o
 22        de otra
 23        :return: String con los cubos
 24        """
 25        caracterAgua = '~'
 26        visualizadorCubos = []
 27
 28        for i in range(1, 9):
 29            if self.aguaEnCubos['8'] < i:
 30                visualizadorCubos.append('      ')
 31            else:
 32                visualizadorCubos.append(caracterAgua * 6)
 33
 34        for i in range(1, 6):
 35            if self.aguaEnCubos['5'] < i:
 36                visualizadorCubos.append('      ')
 37            else:
 38                visualizadorCubos.append(caracterAgua * 6)
 39
 40        for i in range(1, 4):
 41            if self.aguaEnCubos['3'] < i:
 42                visualizadorCubos.append('      ')
 43            else:
 44                visualizadorCubos.append(caracterAgua * 6)
 45
 46        # Devuelve una cadena con los cubos y su contenido de agua
 47        return '''
 48        8|{7}|
 49        7|{6}|
 50        6|{5}|
 51        5|{4}|  5|{12}|
 52        4|{3}|  4|{11}|
 53        3|{2}|  3|{10}|  3|{15}|
 54        2|{1}|  2|{9}|  2|{14}|
 55        1|{0}|  1|{8}|  1|{13}|
 56         +------+   +------+   +------+
 57            8L         5L         3L
 58        '''.format(*visualizadorCubos)

 60    def mostrarEstadoCubos(self):
 61        """
 62        Muestra el objetivo del juego y el estado de los cubos (como de llenos/vacios estan)
 63        :return: String

 69    def checkObjetivo(self):
 70        # Comprueba si uno de los cubos ha conseguido el objetivo
 71        """
 72        Comprueba si has conseguido el objetivo
 73        :return: String
 74        """
 75        for cantidadAgua in self.aguaEnCubos.values():
 76            if cantidadAgua == self.objetivo:
 77                print('Bien hecho! Lo has resuelto en', self.pasos, 'pasos!')

 80    def selecOpcion(self):

104    def selecCubo(self, mensaje):

106        El menu que te dice hacia que cubo quieres dirijir la accion

117    def llenarCubo(self, cuboOrigen):
118        """
119        Llena el cubo seleccionado hasta su tamaño maximo
120        :param cuboOrigen:Char
121        :return: Int
122        """
123        cuboOrigenTam = int(cuboOrigen)
124        self.aguaEnCubos[cuboOrigen] = cuboOrigenTam
125        self.pasos += 1
126
127    def vaciarCubo(self, cuboOrigen):
128        """
129        Vacia el cubo seleccionado
130        :param cuboOrigen: Char
131        :return: nada
132        """
133        self.aguaEnCubos[cuboOrigen] = 0
134        self.pasos += 1
135
136    def moverCubo(self, cuboOrigen, cuboDestino):
137        """
138        Mueve el agua de un cubo a otro
139        :param cuboOrigen: Char
140        :param cuboDestino: Char
141        :return: nada
142        """
143        cuboDestinoTam = int(cuboDestino)
144        espacioVacioCuboDestino = cuboDestinoTam - self.aguaEnCubos[cuboDestino]
145        aguaEnCuboOrigen = self.aguaEnCubos[cuboOrigen]
146        cantidadAMover = min(espacioVacioCuboDestino, aguaEnCuboOrigen)
147
148        # Saco el agua de este cubo
149        self.aguaEnCubos[cuboOrigen] -= cantidadAMover
150
151        # Introduzco el agua en este cubo
152        self.aguaEnCubos[cuboDestino] += cantidadAMover
153        self.pasos += 1
154
155    def jugar(self):
156        """
157        Dependiendo de la opcion que hayas selecionado hace una u otra opcion
158        :return: nada
159        """
160        self.mostrarEstadoCubos()
161        while True:
162            opcion = self.selecOpcion()
163            if opcion == 'L':
164                cubo = self.selecCubo('Selecciona el cubo 8, 5, 3 o SALIR:')
165                self.llenarCubo(cubo)
166            elif opcion == "V":
167                cubo = self.selecCubo('Selecciona el cubo 8, 5, 3 o SALIR:')
168                self.vaciarCubo(cubo)
169            elif opcion == "B":
170                self.vaciarCubo('8')
171                self.vaciarCubo('5')
172                self.vaciarCubo('3')
173            elif opcion == "M":
174                cuboOrigen = self.selecCubo('Selecciona el cubo ORIGEN 8, 5, 3 o SALIR:')
175                cuboDestino = self.selecCubo('Selecciona el cubo DESTINO 8, 5, 3 o SALIR:')
176                self.moverCubo(cuboOrigen, cuboDestino)
177            elif opcion == "T":
178                self.llenarCubo('8')
179                self.llenarCubo('5')
180                self.llenarCubo('3')
181            self.mostrarEstadoCubos()
182            self.checkObjetivo()
